87 void tb3::go_to_goal_callback() {
88  if ((!m_go_to_goal) && (!m_go_to_angle))
89  if (goals.size() != 0) {
90  set_goal(goals);
91  } else {
92  return;
93  }
94 
95  if (m_go_to_goal == false && m_go_to_angle == true) {
96  double m_current_orientation = compute_yaw_from_quaternion();
97  double angle_to_goal = (m_goal_orientation - m_current_orientation);
98 
99  if (angle_to_goal < 0)
100  angle_to_goal = normalize_angle_positive(angle_to_goal);
101 
102  // angle to rotate to face the goal
103  double w =
104  angle_to_goal; // m_goal_orientation - compute_yaw_from_quaternion();
105  if (w > M_PI) {
106  w = w - 2 * M_PI;
107  // w = m_normalize_angle_positive(w);
108  }
109 
110  w = abs(w);
111  // proportional control for angular velocity
112  double angular_z = m_kh * w;
113  if (angular_z > 0)
114  angular_z = std::min(angular_z, m_angular_speed);
115  else
116  angular_z = std::max(angular_z, -m_angular_speed);
117 
118  if (w > 0.05) {
119  // proportional control for linear velocity
120  double linear_x = 0.0;
121  move_tb3(linear_x, angular_z);
122  } else {
123  m_go_to_angle = false;
124  stop();
125  }
126  }
127 
128  if (m_go_to_goal) {
129  std::pair<double, double> goal{m_goal_x, m_goal_y};
130  double distance_to_goal = compute_distance(m_location, goal);
131 
132  if (distance_to_goal > 0.1) {
133  distance_to_goal = compute_distance(m_location, goal);
134  double angle_to_goal =
135  std::atan2(m_goal_y - m_location.second, m_goal_x - m_location.first);
136 
137  if (angle_to_goal < 0)
138  // angle_to_goal = 2 * M_PI + angle_to_goal;
139  angle_to_goal = normalize_angle_positive(angle_to_goal);
140 
141  // angle to rotate to face the goal
142  double w = angle_to_goal - compute_yaw_from_quaternion();
143  if (w > M_PI) {
144  w = w - 2 * M_PI;
145  // w = m_normalize_angle_positive(w);
146  }
147 
148  // proportional control for linear velocity
149  double linear_x = std::min(m_kv * distance_to_goal, m_linear_speed);
150 
151  // proportional control for angular velocity
152  double angular_z = m_kh * w;
153  if (angular_z > 0)
154  angular_z = std::min(angular_z, m_angular_speed);
155  else
156  angular_z = std::max(angular_z, -m_angular_speed);
157  move_tb3(linear_x, angular_z);
158  } else {
159  // RCLCPP_INFO_STREAM(this->get_logger(), "********** Goal reached
160  // **********"); stop();
161  m_go_to_goal = false;
162  stop();
163  }
164  }
165 }
