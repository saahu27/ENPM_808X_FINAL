30 void MinimalSubscriber::topic_callback(
31  const tutorial_interfaces::msg::Num::SharedPtr msg) {
32  if (abs(static_cast<float>(ang)) <= 0.0001 &&
33  abs(static_cast<float>(lin_x)) <= 0.0001 &&
34  abs((static_cast<float>(lin_y))) <= 0.0001) {
35  for (size_t i = 0; i < msg->poses.size(); i++) {
36  if (flag[msg->marker_ids[i]] != -1) {
37  auto pose = msg->poses[i];
38 
39  x[msg->marker_ids[i]] = pose.position.x;
40  y[msg->marker_ids[i]] = pose.position.y;
41  z[msg->marker_ids[i]] = pose.position.z;
42 
43  id[msg->marker_ids[i]]++;
44  RCLCPP_INFO(this->get_logger(), "Product 1 count: '%d'", id[0]);
45  RCLCPP_INFO(this->get_logger(), "Product 2 count: '%d'", id[1]);
46  RCLCPP_INFO(this->get_logger(), "Product 3 count: '%d'", id[2]);
47  RCLCPP_INFO(this->get_logger(), "Product 4 count: '%d'", id[3]);
48 
49  x_prev[msg->marker_ids[i]] = x[msg->marker_ids[i]];
50  y_prev[msg->marker_ids[i]] = y[msg->marker_ids[i]];
51  z_prev[msg->marker_ids[i]] = z[msg->marker_ids[i]];
52 
53  if (i == msg->poses.size() - 1) {
54  flag[msg->marker_ids[i]] = -1;
55  }
56  }
57  }
58  }
